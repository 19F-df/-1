    <script>
        let customers = [];
        let schedules = [];

        const contentTemplates = {
            marketing: "这辆{model}真的太帅了！😱 外观超吸睛,开出去回头率绝对爆表！内饰科技感满满,大屏幕超清晰。动力也够劲,起步轻快,超车毫无压力。关键是性价比超高,20万以内就能拿下！想知道具体配置吗？评论区问我👇",
            promotion: "{model}限时特惠！💰 直降3万！仅限本周！前50名订车再送终身保养！名额有限,先到先得！感兴趣的赶紧私信我咨询📱",
            qa: "很多人问{model}油耗怎么样？今天实测一下！市区走走停停,百公里才7个油！高速更省,不到6个！这个级别真的很省了。大家还想了解什么？评论区告诉我！",
            review: "提{model}一个月了,说说真实感受。优点:空间够大,一家四口够用；辅助驾驶好用,高速很轻松。缺点:隔音一般,胎噪有点大。总体很满意,20万这个价位很值！你们觉得呢？",
            hot_topic: "继小米汽车之后,国产车又出爆款了！{model}这次真的杀疯了！外观内饰全面升级,关键价格还降了！国产车真的支棱起来了！"
        };

        const hashtags = {
            marketing: ['#汽车推荐', '#买车攻略', '#新车上市', '#试驾体验'],
            promotion: ['#汽车优惠', '#购车福利', '#限时特惠'],
            qa: ['#汽车问答', '#用车小知识', '#实测'],
            review: ['#真实车主', '#用车感受', '#测评'],
            hot_topic: ['#汽车热点', '#热门', '#国产汽车']
        };

        function showSection(section) {
            document.querySelectorAll('.main-section').forEach(el => el.classList.add('hidden'));
            document.getElementById(section + '-section').classList.remove('hidden');
        }

        function generateContent() {
            const carModel = document.getElementById('carModel').value || '比亚迪汉';
            const contentType = document.getElementById('contentType').value;
            
            let content = contentTemplates[contentType].replace(/{model}/g, carModel);
            const tagList = hashtags[contentType];
            const tagHtml = tagList.map(tag => `<span class="hashtag">${tag}</span>`).join('');
            
            document.getElementById('contentResult').style.display = 'block';
            document.getElementById('contentText').textContent = content;
            document.getElementById('hashtags').innerHTML = tagHtml;
        }

        function addCustomer() {
            const name = document.getElementById('customerName').value;
            const phone = document.getElementById('customerPhone').value;
            const model = document.getElementById('interestedModel').value || '未指定';
            const intent = document.getElementById('intentLevel').value;
            
            if (!name || !phone) {
                alert('请填写姓名和电话');
                return;
            }
            
            customers.push({ name, phone, model, intent });
            updateCustomerList();
            alert('客户添加成功！');
            
            document.getElementById('customerName').value = '';
            document.getElementById('customerPhone').value = '';
            document.getElementById('interestedModel').value = '';
        }

        function updateCustomerList() {
            const tbody = document.getElementById('customerList');
            tbody.innerHTML = customers.map(c => {
                const intentText = c.intent === 'high' ? '高意向' : c.intent === 'medium' ? '中意向' : '低意向';
                const intentClass = 'status-' + c.intent;
                return `<tr>
                    <td>${c.name}</td>
                    <td>${c.phone}</td>
                    <td>${c.model}</td>
                    <td><span class="status-badge ${intentClass}">${intentText}</span></td>
                </tr>`;
            }).join('');
            
            document.getElementById('customerResult').style.display = 'block';
        }

        function generateVideo() {
            const script = document.getElementById('videoScript').value;
            if (!script) {
                alert('请输入文案内容');
                return;
            }
            document.getElementById('videoResult').style.display = 'block';
            document.getElementById('videoText').textContent = '✨ 视频生成功能\n\n文案: ' + script + '\n\n(演示版本,完整版可集成AI视频API)';
        }

        function addSchedule() {
            const content = document.getElementById('scheduleContent').value;
            const time = document.getElementById('scheduleTime').value;
            
            if (!content || !time) {
                alert('请填写内容和发布时间');
                return;
            }
            
            schedules.push({ content, time });
            updateScheduleList();
            alert('排期添加成功！');
        }

        function updateScheduleList() {
            const listDiv = document.getElementById('scheduleList');
            listDiv.innerHTML = schedules.map((s, i) => `
                <div style="padding: 10px; margin-bottom: 10px; background: #f5f5f5; border-radius: 8px;">
                    <strong>排期 ${i + 1}</strong><br>
                    ⏰ 时间: ${s.time}<br>
                    📝 内容: ${s.content.substring(0, 50)}...
                </div>
            `).join('');
            
            document.getElementById('scheduleResult').style.display = 'block';
        }
    </script>
